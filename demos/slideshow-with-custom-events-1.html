Code

JavaScript


var myCustomEvent = 'button:clicked';

new SlideShow('listOfImages', { events: { play: 'button:clicked' } });

document.observe('dom:loaded', function(){
  $('startButton').observe('click', function(e){
    e.element().fire(myCustomEvent);
    e.element().insert({ after: '<p>' + myCustomEvent + '</p>' }).remove();
  });
});
